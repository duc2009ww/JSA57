while True: try: n = int(input("Nhập một số nguyên dương n: ")) if n > 0: break else: print("Vui lòng nhập số nguyên dương.") except ValueError: print("Đầu vào không hợp lệ. Vui lòng nhập một số nguyên.") sum_even = 0 sum_odd = 0 for i in range(1, n + 1): if i % 2 == 0: sum_even += i else: sum_odd += i print(f"Tổng các số chẵn từ 1 đến {n} là: {sum_even}") print(f"Tổng các số lẻ từ 1 đến {n} là: {sum_odd}")
